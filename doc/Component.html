# File component.rb, line 19
def initialize name, id, spprice=0
    if id == 0
        @price = SkillPointPrice.new spprice
        @img = ''
    else
        @price = GoldPrice.new id
        @img = self.getByID(id)['img']
    end
    @id = id
    @name = name
end
